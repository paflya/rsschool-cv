        public ArrayList ExtractPrepods(Document Xml) {
            ArrayList PrepList=new ArrayList();
            try {
                Node root = Xml.getDocumentElement();
                NodeList AllINodes = root.getChildNodes();
                int PrepodsIndex=0;
                for (int i = 0; i < AllINodes.getLength(); i++) {
                    if (AllINodes.item(i).getNodeName().equals("prepods")) {
                        PrepodsIndex = i;
                        break;
                    }
                }
                NodeList prepods = AllINodes.item(PrepodsIndex).getChildNodes();
                for (int i = 0; i < prepods.getLength(); i++) {
                    Node Prep = prepods.item(i);
                    if (Prep.getNodeType() == Node.ELEMENT_NODE)
                        PrepList.add(Prep.getChildNodes().item(1).getTextContent());
                }
                return PrepList;
            }
            catch (Exception ex)
            {
                Log.d("Err",ex.toString());
                return PrepList;
            }
        }
